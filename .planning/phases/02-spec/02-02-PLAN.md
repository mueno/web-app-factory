From tools/phase_executors/base.py:
```python
class PhaseExecutor(ABC):
    @property
    @abstractmethod
    def phase_id(self) -> str: ...
    @property
    @abstractmethod
    def sub_steps(self) -> list: ...
    @abstractmethod
    def execute(self, ctx: PhaseContext) -> PhaseResult: ...

@dataclass(frozen=True)
class PhaseContext:
    run_id: str; phase_id: str; project_dir: Path; idea: str; app_name: str
    extra: dict = field(default_factory=dict)

@dataclass
class PhaseResult:
    phase_id: str; success: bool; artifacts: list; sub_steps: list
    error: str | None = None; resume_point: str | None = None

@dataclass
class SubStepResult:
    sub_step_id: str; success: bool; artifacts: list
    error: str | None = None; notes: str = ""
```

    ```
    class Phase1aSpecExecutor(PhaseExecutor):
        phase_id = "1a"
        sub_steps = ["research", "analyze", "write_validation", "write_feasibility", "self_assess"]

        def execute(self, ctx: PhaseContext) -> PhaseResult:
            1. Load quality criteria for "1a" from YAML contract
               (contract_path from ctx.extra.get("contract_path") with fallback to default)
            2. Build phase-specific system prompt from SPEC_AGENT.system_prompt + quality criteria
            3. Construct user prompt incorporating ctx.idea and ctx.app_name
               - Prompt instructs agent to:
                 a. Research competitors via WebSearch
                 b. Analyze market fit and target user
                 c. Write idea-validation.md with structured sections
                 d. Write tech-feasibility-memo.json with rendering strategy evaluation
                 e. Include explicit go_no_go: Go or go_no_go: No-Go
            4. Call run_spec_agent() with the prompts
            5. Parse agent output:
               - If agent wrote files directly (via Write tool): verify they exist
               - If agent returned text: parse and write files
            6. Validate tech-feasibility-memo.json:
               - Parse JSON
               - Extract package names
               - Call validate_npm_packages() to check each against registry
               - Mark invalid packages in the memo
            7. Generate quality self-assessment (CONT-04)
            8. Return PhaseResult with artifacts list

        - _validate_npm_packages_in_memo(memo_path) -> dict[str, bool]
    ```
