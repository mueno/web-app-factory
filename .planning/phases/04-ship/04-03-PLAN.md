class PhaseContext:

    project_dir: Path
    idea: str

class SubStepResult:

    success: bool
    artifacts: list
    error: str | None
    notes: str

class PhaseResult:
    success: bool
    sub_steps: list[SubStepResult]
    error: str | None
    notes: str

class PhaseExecutor(ABC):
    @property
