class PhaseContext:
    run_id: str
    phase_id: str
    project_dir: Path
    idea: str
    app_name: str
    extra: dict  # {"company_name": str|None, "contact_email": str|None}

class SubStepResult:
    sub_step_id: str
    success: bool
    artifacts: list
    error: str | None
    notes: str

class PhaseResult:
    success: bool
    sub_steps: list[SubStepResult]
    error: str | None
    notes: str

class PhaseExecutor(ABC):
    @property
    def phase_id(self) -> str: ...
    @property
    def sub_steps(self) -> list: ...
    def execute(self, ctx: PhaseContext) -> PhaseResult: ...
