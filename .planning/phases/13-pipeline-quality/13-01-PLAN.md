<objective>
Split the Phase 2b build executor from a single monolithic agent call into three checkpointed sub-steps (shared components, pages, integration), each with a focused prompt and individual failure/resume capability.

Purpose: When the build agent fails on page generation, the user currently must restart from scratch. Sub-step decomposition allows resumption from the failed step and produces clearer error diagnostics. The focused prompts also improve generation quality by reducing prompt size per agent call.

Output: Modified `phase_2b_executor.py` with three `run_build_agent` calls, updated tests covering checkpoints, resume, and prompt isolation.
</objective>

```python
class PhaseContext:
    run_id: str
    phase_id: str
    project_dir: Path
    idea: str

    budget: float = 50.0
    listing_mode: str = "direct"

@dataclass
class SubStepResult:
    sub_step_id: str
    success: bool
    artifacts: list = field(default_factory=list)
    error: str | None = None
    notes: str = ""

@dataclass
class PhaseResult:
    phase_id: str
    success: bool
    artifacts: list = field(default_factory=list)
    sub_steps: list = field(default_factory=list)
    error: str | None = None
    resume_point: str | None = None

class PhaseExecutor(ABC):
    def _start_index(self, ctx: PhaseContext) -> int:
        target = ctx.resume_sub_step
        if target is None:
            return 0
        if target not in self.sub_steps:
            return 0
        return self.sub_steps.index(target)
```
